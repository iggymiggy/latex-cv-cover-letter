% Global document settings (shared by CV and cover letter)
%
% Helper macro for flexible file loading
% Works when compiling from root, templates/, or companies/<name>/
% Tries multiple path locations to find files
\newcommand{\loadfile}[2]{%
  \IfFileExists{../../#1}{\input{../../#1}}{%
  \IfFileExists{../#1}{\input{../#1}}{%
  \IfFileExists{#1}{\input{#1}}{#2}}}%
}

% Paper size (default: A4)
% - Use: a4paper (most of the world)
% - Use: letterpaper (US/Canada)
%
% Change the value below and recompile.
\providecommand{\papersize}{a4paper}

% Font size (default: 11pt)
% - Use: 10pt, 11pt, 12pt
\providecommand{\docfontsize}{11pt}

% Pass paper size option to the article class (must happen before \documentclass)
\expandafter\PassOptionsToClass\expandafter{\papersize}{article}
\expandafter\PassOptionsToClass\expandafter{\docfontsize}{article}
